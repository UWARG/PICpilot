194  while (U1STAbits.URXDA) {
195  data = U1RXREG;
196 
197  if (data == '$') {

201  } else if (data == 0x0A) {
202  //End of Packet
203  stringType = 0;
204  byteCount = 0;
205  newData = 1;
206  } else if (stringType == 1){
207  if (!readLock)
208  gga[byteCount] = data;
209  } else if (stringType == 2){
210  if (!readLock)
211  vtg[byteCount] = data;

213 
214 
215  if (byteCount <= 5){
216  headerData[byteCount] = data;
217  }
218 
219  //Identify the packet type once the header has been read
220  if (byteCount == 5) {
221  stringType = identifyString(&headerData);
222  }
223  byteCount++;
224  }
225  IFS0bits.U1RXIF = 0; // Clear the Interrupt Flag
226 }
