28 /* Buffer used for SPI read and write responses */
29 /* Both the read and write register SPI routines below use this packet
30  to store the returned SPI response. None of the write register commands
31  implemented in this library check the data that is returned by the sensor
32  to ensure that it is consistent with the data that was sent. For normal
33  cases this isn't necessary however if you wish to implement your own
34  checking then this is the structure that you need to check after each
35  register set command. The structure has the following form:
36  VN_SPI_LastReceivedPacket.CmdID -> This is the ID for the command that
37  the response is for
38  VN_SPI_LastReceivedPacket.RegID -> This is the ID for the register that
39  the response is for
40  VN_SPI_LastReceivedPacket.Data[] -> This is the data that was returned by
41  the sensor as an array of unsigned 32-bit
42  integers */
43 VN100_SPI_Packet VN_SPI_LastReceivedPacket = {0, 0, 0, 0, {{0}}};

48 * Function Name : VN100_initSPI()
49 * Description : Initialize the SPI interface.
50 * Input : None
51 * Output : None
52 * Return : None
53 *******************************************************************************/
54 void VN100_initSPI(){
55  //Set interrupts
56  IFS2bits.SPI2IF = 0; //Clear interrupt flag
57  IEC2bits.SPI2IE = 0; //Disable interrupt (so it doesnt mess with this initialization)
58  //Continue module operation in idle mode
59  //SPI2STATbits.SPISIDL = 0; commented out as it does not exist in Trevor's past code
60  //SPI clock controlled by this module
61  SPI2CON1bits.DISSCK = 0;
62  //Output pins are controlled by this module
63  SPI2CON1bits.DISSDO = 0;
64  //16/8 bit communication mode (1/0)
65  SPI2CON1bits.MODE16 = 0; //8
66  //Master mode(1)/Slave mode(0)
67  SPI2CON1bits.MSTEN = 1; //MASTER
68  //Sample Phase (end/middle)
69  SPI2CON1bits.SMP = 0; //Sample the input at the middle of the square wave
70  //Clock Edge Select
71  SPI2CON1bits.CKE = 0; //Output data changes from idle state to active clock state (1 is the opposite)
72  //Clock Polarity
73  SPI2CON1bits.CKP = 1; //Idle clock state is high, active clock state is low
74  //Secondary Prescale (The prescale of the prescale)(3 bits)
75  SPI2CON1bits.SPRE = 0b110; //8:1 prescale
76  //Primart Prescale (The prescale of the clock) (2 bits)
77  SPI2CON1bits.PPRE = 0b10; //64:1 prescale
78 
79  //Then enable interrupts
80  IFS2bits.SPI2IF = 0; //Clear interrupt flag
81  IEC2bits.SPI2IE = 0; //Disable interrupt
82 
83  SPI2STATbits.SPIROV = 1;
84  //Enable SPI
85  SPI2STATbits.SPIEN = 1;
86 
87  TRISGbits.TRISG9 = 0;
88 }

90 * Function Name : VN100_SPI_ReadRegister(unsigned char sensorID, unsigned char regID, unsigned char regWidth)
91 * Description : Read the register with the ID regID on a VN-100 sensor
92 * using the SPI interface.
93 * Input : sensorID -> The sensor to get the requested data from.
94 * : regID -> The requested register ID number
95 * : regWidth -> The width of the requested register in 32-bit words
96 * Output : None
97 * Return : Pointer to SPI packet returned by the sensor
98 *******************************************************************************/
99 VN100_SPI_Packet* VN100_SPI_ReadRegister(unsigned char sensorID, unsigned char regID, unsigned char regWidth){
100 
101 
102  /* Pull SS line low to start transaction*/
103  VN_SPI_SetSS(sensorID, VN_PIN_LOW);
104  /* Send request */
105  VN_SPI_SendReceive(VN_BYTES2WORD(0, 0, regID, VN100_CmdID_ReadRegister));
106  VN_SPI_SendReceive(0);
107  /* Pull SS line high to end SPI transaction */
108  VN_SPI_SetSS(sensorID, VN_PIN_HIGH);
109  /* Delay for 50us */
110  //TODO: See if we need this. If we're running out of time, leverage this however we can
111  VN_Delay(50);
112  /* Pull SS line low to start SPI transaction */
113  VN_SPI_SetSS(sensorID, VN_PIN_LOW);
114 
115  /* Get response over SPI */
116  unsigned long i;
117  for(i=0;i<=regWidth;i++){
118  *(((unsigned long*)&VN_SPI_LastReceivedPacket) + i) = VN_SPI_SendReceive(0);
119  }
120 //
121 // /* Pull SS line high to end SPI transaction */
122  VN_SPI_SetSS(sensorID, VN_PIN_HIGH);
123  VN_Delay(50);
124 // /* Return Error code */
125  return &VN_SPI_LastReceivedPacket;
126 }

130 * Description : Write to the register with the ID regID on VN-100 sensor
131 * using the SPI interface.
132 * Input : sensorID -> The sensor to write the requested data to.
133 * : regID -> The register ID number
134 * : regWidth -> The width of the register in 32-bit words
135 * Output : ptrWriteValues -> The data to write to the requested register.
136 * Return : Pointer to SPI packet returned by the sensor
137 *******************************************************************************/
138 VN100_SPI_Packet* VN100_SPI_WriteRegister(unsigned char sensorID, unsigned char regID, unsigned char regWidth, unsigned long* ptrWriteValues){
139 
140  unsigned long i;
141  /* Pull SS line low to start transaction*/
142  VN_SPI_SetSS(sensorID, VN_PIN_LOW);
143  /* Send write command */
144  VN_SPI_SendReceive(VN_BYTES2WORD(0, 0, regID, VN100_CmdID_WriteRegister));
145  for(i=0;i<regWidth;i++){
146  VN_SPI_SendReceive(ptrWriteValues[i]);
147  }
148 
149  /* Pull SS line high to end SPI transaction */
150  VN_SPI_SetSS(sensorID, VN_PIN_HIGH);
151 
152  /* Delay for 50us */
153  VN_Delay(100);
154 
155  /* Pull SS line low to start SPI transaction */
156  VN_SPI_SetSS(sensorID, VN_PIN_LOW);
157 
158  /* Get response over SPI */
159  for(i=0;i<4;i++){
160  *(((unsigned long*)&VN_SPI_LastReceivedPacket) + i) = VN_SPI_SendReceive(0);
161  }
162 
163  /* Pull SS line high to end SPI transaction */
164  VN_SPI_SetSS(sensorID, VN_PIN_HIGH);
165 
166 
167  /* Return pointer to SPI packet */
168  return &VN_SPI_LastReceivedPacket;
169 }
