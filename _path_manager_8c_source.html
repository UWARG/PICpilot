442 float followOrbit(float* center, float radius, char direction, float* position, float heading){//Heading in degrees (magnetic)
443  heading = deg2rad(90 - heading);
444 
445 
446  float orbitDistance = sqrt(pow(position[0] - center[0],2) + pow(position[1] - center[1],2));
447  float courseAngle = atan2(position[1] - center[1], position[0] - center[0]); // (y,x) format
448 
449  while (courseAngle - heading < -PI){
450  courseAngle += 2 * PI;
451  }
452  while (courseAngle - heading > PI){
453  courseAngle -= 2 * PI;
454  }
455 
456  return 90 - rad2deg(courseAngle + direction * (PI/2 + atan(k_gain[ORBIT] * (orbitDistance - radius)/radius))); //Heading in degrees (magnetic)

458 float followStraightPath(float* waypointDirection, float* targetWaypoint, float* position, float heading){ //Heading in degrees (magnetic)
459  heading = deg2rad(90 - heading);//90 - heading = magnetic heading to cartesian heading
460  float courseAngle = atan2(waypointDirection[1], waypointDirection[0]); // (y,x) format
461  while (courseAngle - heading < -PI){
462  courseAngle += 2 * PI;
463  }
464  while (courseAngle - heading > PI){
465  courseAngle -= 2 * PI;
466  }
467

469  float calcHeading = 90 - rad2deg(courseAngle - MAX_PATH_APPROACH_ANGLE * 2/PI * atan(k_gain[PATH] * pathError)); //Heading in degrees (magnetic)
470  return calcHeading;
