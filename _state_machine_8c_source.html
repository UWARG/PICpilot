35 void StateMachine(char entryLocation){
36  //Timers
37  dTime = (int)(getTime() - stateMachineTimer);
38  stateMachineTimer = getTime();
39  uplinkTimer += dTime;
40  downlinkP0Timer += dTime;
41  downlinkP1Timer += dTime;
42  downlinkP2Timer += dTime;
43  stateMachineTimer += dTime;
44  imuTimer += dTime;
45  dmaTimer += dTime;
46 
47 
48  //Clear Watchdog timer
49  asm("CLRWDT");
50  //Feedback systems such as this autopilot are very sensitive to timing. In order to keep it consistent we should try to keep the timing between the calculation of error corrections and the output the same.
51  //In other words, roll pitch and yaw control, mixing, and output should take place in the same step.
52  if(AMUpdate){
53  AMUpdate = 0;
54  //Run - Angle control, and angular rate control
55  //Input from Controller
56  inputCapture();
57  //Recalculate all data dependent on any DMA data
58  highLevelControl();
59  lowLevelControl();
60  }
61  else if(IMU_UPDATE_FREQUENCY <= imuTimer && entryLocation != STATEMACHINE_IMU){
62 // debug("IMU");
63  imuTimer = 0;
64  //Poll Sensor
65  imuCommunication();
66  //Input from Controller
67  inputCapture();
68  highLevelControl();
69  lowLevelControl();
70 
71  }
72  else if(isDMADataAvailable() && checkDMA()){
73  //Input from Controller
74  inputCapture();
75  //Recalculate all data dependent on any DMA data
76  highLevelControl();
77  lowLevelControl();
78  }
79  else{
80  }
81 
82  if(UPLINK_CHECK_FREQUENCY <= uplinkTimer){
83  uplinkTimer = 0;
84  readDatalink();
85  }
86 
87  if(P0_SEND_FREQUENCY <= downlinkP0Timer){
88 // debug("P0");
89  //Compile and send data
90  downlinkP0Timer = 0;
91  writeDatalink(PRIORITY0);
92  }
93  else if(P1_SEND_FREQUENCY <= downlinkP1Timer){
94 // debug("P1");
95  //Compile and send data
96  downlinkP1Timer = 0;
97  writeDatalink(PRIORITY1);
98  }
99  else if(P2_SEND_FREQUENCY <= downlinkP2Timer || areGainsUpdated()){
100 // debug("P2");
101  //Compile and send data
102  downlinkP2Timer = 0;
103  writeDatalink(PRIORITY2);
104  }
105  else{
106  //Then Sleep
107  }
108  //Loop it back again!
109  inboundBufferMaintenance();
110  outboundBufferMaintenance();
111  asm("CLRWDT");
112 }
